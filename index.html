4.7: JS Form Validation

HTML Syntax

<form name="jsFormValidate" onsubmit="return js_validateForm()">
    <label for="jsv-name">Name</label>
    <input type="text" name="name" id="jsv-name">

    <label for="jsv-num">Number</label>
    <input type="text" name="num" id="jsv-num">
    <span id="jsv-numloc" class="form-error"></span>
    
    <button type="submit">Validate</button>
</form>

JavaScript Syntax (from script.js)

function js_validateForm() {
    let name = document.jsFormValidate.name.value;
    let num = document.jsFormValidate.num.value;
    let numLoc = document.getElementById("jsv-numloc");
    
    numLoc.innerHTML = ""; // Clear old error

    if (name == null || name == "") {
        showMessage("Name can't be blank");
        return false;
    }
    if (num != "" && isNaN(num)) {
        numLoc.innerHTML = "Enter Numeric value only";
        return false;
    }
    
    showMessage("Form is valid!");
    return false; // Prevent submission
}

Output

4.8: DOM Manipulation

HTML Syntax

<h4 id="js-dom-header" onclick="js_getElement()">
    <code>getElementById()</code> (Click Me)
</h4>

<button onclick="js_createElement()">Add a new <p> element</button>
<div id="dom-creation-target"></div>

JavaScript Syntax (from script.js)

function js_getElement() {
    var x = document.getElementById("js-dom-header");
    showMessage("I am a " + x.tagName + " element");
}

function js_createElement() {
    const para = document.createElement("p");
    para.innerText = "Dynamically created paragraph.";
    para.style.color = "#16a34a";
    document.getElementById("dom-creation-target").appendChild(para);
}

Output

getElementById() (Click Me)

createElement()
